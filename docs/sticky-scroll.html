	$('.stickyBar').stickyOnScroll({
		stoppedBy: $('.appbar')
	});

	$('.stickyBar2').stickyOnScroll({
		stoppedBy: $('.stickyBar')
	});
});

    Parameters

    stuckClass
    The CSS class to add when an element is in th stuck state, defaults to stuck.
    handler
    The function to call when crossing the boundary, the 1st parameter is the direction of crossing up or down.
    offset
    Offset to apply to the stickiness in pixels.
    stoppedBy
    The element which this element will stop under.
