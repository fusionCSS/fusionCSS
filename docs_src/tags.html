Tag Input

To use tag input functionality the fusionCSS.js and fusionCSSTagInput.js libraries needs to be included.

<input id="tagsInput1" data-delimiter="," data-placeholder="Add Tags" type="text" value="foo,bar" />
<input id="tagsInput2" data-delimiter="." data-placeholder="Add More Tags" type="text" value="" />
$('#tagsInput1').enableTagInput();
$('#tagsInput2').enableTagInput({
	tagHTML: '<div class="chip tag"><b class="primary">T</b><span class="tagValue">{value}</span><i class="tagDelete">x</i></div>'
});
